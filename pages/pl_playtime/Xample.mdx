server.lua
```lua
--/testquest <questId> [amount]
RegisterCommand('testquest', function(source, args)
    local questId = tonumber(args[1])
    local amount = tonumber(args[2]) or 1
    
    if not questId then return end
    
    exports['pl_playtime']:UpdateQuestProgress(source, questId, amount)
    
end, false)
```
