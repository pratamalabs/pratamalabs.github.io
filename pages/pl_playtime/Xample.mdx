server.lua
```lua
--/testquest <missionID> [amount]
RegisterCommand('testquest', function(source, args)
    local missionID = tonumber(args[1])
    local amount = tonumber(args[2]) or 1
    
    if not missionID then return end
    
    exports['pl_playtime']:UpdateMissionProgress(source, missionID, amount)
    
end, false)
```
